


DROP DATABASE IF EXISTS feinstaupmessung;
CREATE DATABASE IF NOT EXISTS feinstaupmessung;



CREATE TABLE sensor (
sensor_id INTEGER NOT NULL PRIMARY KEY,
sensor_typ TEXT,
); 

CREATE TABLE standort(
standort_id INTEGER PRIMARY KEY,
standort TEXT,breitegrad INTEGER,
laengegrad INTEGER
); 

CREATE TABLE feinstaubsensor(
sds11_id INTEGER PRIMARY KEY,
zeitstempel REAL,
p1 REAL,
durP1 REAL,
ratioP1 REAL,
p2 REAL,
durP2 REAL,
ratioP2 REAL,
FOREIGN KEY (sensor_id) REFERENCES sensor (sensor_id),
FOREIGN KEY (standort_id) REFERENCES standort (standort_id)
); 

CREATE TABLE tfsensor(
dht22_id INTEGER PRIMARY KEY,
luftfeuchtigkeit REAL,
temperatur REAL,
zeitstempel TEXT,
FOREIGN KEY (sensor_id) REFERENCES sensor (sensor_id),
FOREIGN KEY (standort_id) REFERENCES standort (standort_id)
);
